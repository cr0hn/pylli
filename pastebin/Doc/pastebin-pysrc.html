327          """ 
328          Submit a code snippet to Pastebin. 
329   
330          @type  paste_code: str 
331          @param paste_code: Code or text to send to U{http://pastebin.com}. 
332   
333          @type  paste_name: str 
334          @param paste_name: (Optional) Name of the author of the paste. 
335              Default is to paste anonymously. 
336   
337          @type  paste_private: bool 
338          @param paste_private: (Optional) C{True} if the paste is private (only 
339              visible with the link), C{False} if it's public (indexed and 
340              searchable). The Pastebin FAQ (U{http://pastebin.com/faq}) claims 
341              private pastes are not indexed by search engines (aka Google). 
342   
343          @type  paste_expire_date: str 
344          @param paste_expire_date: (Optional) Expiration date for the paste. 
345              Once past this date the paste is deleted automatically. Valid 
346              values are found in the L{Pastebin.paste_expire_date} class member. 
347              If not provided, the paste never expires. 
348   
349          @type  paste_format: str 
350          @param paste_format: (Optional) Programming language of the code being 
351              pasted. This enables syntax highlighting when reading the code in 
352              U{http://pastebin.com}. Default is no syntax highlighting (text is 
353              just text and not source code). 
354   
355          @rtype:  str 
356          @return: Returns the URL to the newly created paste. 
357   
358          @raise PastebinError: The Pastebin API has returned an error message. 
359   
360          @raise IOError: An error occurred when contacting the Pastebin web 
361              application. This is typically caused by network errors. 
362          """ 
363   
364          # Code snippet to submit 
365          argv = { 'paste_code' : str(paste_code) } 
366   
367          # Name of the poster 
368          if paste_name is not None: 
369              argv['paste_name'] = str(paste_name) 
370   
371          # Is the snippet private? 
372          if paste_private is not None: 
373              argv['paste_private'] = int(bool(int(paste_private))) 
374   
375          # Expiration for the snippet 
376          if paste_expire_date is not None: 
377              paste_expire_date = str(paste_expire_date).strip().upper() 
378              argv['paste_expire_date'] = paste_expire_date 
379   
380          # Syntax highlighting 
381          if paste_format is not None: 
382              paste_format = str(paste_format).strip().lower() 
383              argv['paste_format'] = paste_format 
384   
385          # Make the request to the Pastebin API 
386          fd = urllib.urlopen(cls._api_url, urllib.urlencode(argv)) 
387          try: 
388              response = fd.read() 
389          finally: 
390              fd.close() 
391          del fd 
392   
393          # Return the new snippet URL on success, raise exception on error 
394          if not response.startswith(cls._prefix_url): 
395              raise PastebinError(response) 
396          return response
